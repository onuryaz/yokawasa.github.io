Go by Example: Errors

Goでは、明示的な別の戻り値を使用してエラーを伝えることは一般的です。 これは、JavaやRubyなどの言語で使用される例外やC言語で時折使用される単一の結果やエラー値とは対照的です。 Goのアプローチは、どの関数がエラーを返すのかを理解しやすくし、 それらを他のエラーのないタスクに使用されているのと同じ言語構造で処理できるようにすることです。

 
package main

import "errors"
import "fmt"

通常、エラーは最後の戻り値であり、組み込みのインタフェースであるerror型を持っています。

 
func f1(arg int) (int, error) {
    if arg == 42 {

errors.Newは、与えられたエラーメッセージを元に基本的なerror値を作成します。

 
        return -1, errors.New("can't work with 42")
    }

エラーの場所にnil値を設定することはエラーがなかったことを示します。

 
    return arg + 3, nil
}

Error()メソッドを実装してカスタムエラーとして使用することは可能です。 これは、明示的に引数エラーを表すためにカスタムタイプを使用している例になります。

 
type argError struct {
    arg  int
    prob string
}

func (e *argError) Error() string {
    return fmt.Sprintf("%d - %s", e.arg, e.prob)
}

func f2(arg int) (int, error) {
    if arg == 42 {

この場合、新しい構造体を構築するために&argError構文を使用して、argとprobの2つのフィールドに値を与えています。

 
        return -1, &argError{arg, "can't work with it"}
    }
    return arg + 3, nil
}

以下の2つのループでは、それぞれのエラーを返す関数をテストします。 if行におけるインラインでのエラーチェックは、Goコードにおける共通イディオムであることをおしりおきください。

 
    for _, i := range []int{7, 42} {
        if r, e := f1(i); e != nil {
            fmt.Println("f1 failed:", e)
        } else {
            fmt.Println("f1 worked:", r)
        }
    }
    for _, i := range []int{7, 42} {
        if r, e := f2(i); e != nil {
            fmt.Println("f2 failed:", e)
        } else {
            fmt.Println("f2 worked:", r)
        }
    }

カスタムエラーにおいてプログラムでデータを使用したい場合は、 エラーを型アサーションを介してカスタムエラータイプのインスタンスとして取得する必要があります。

 
    _, e := f2(42)
    if ae, ok := e.(*argError); ok {
        fmt.Println(ae.arg)
        fmt.Println(ae.prob)
    }
}
